使用說明
screen.height; // returns height of screen
$(window).height(); // returns height of browser viewport
$(document).height(); // returns height of HTML document
$('#box').innerHeight(); // element height (含padding 不包含 border margin)
$('#box').outerHeight(); // element height (含padding 及border ，不包含 margin)
$('#box').outerHeight(true); // element height (含padding border margin)
